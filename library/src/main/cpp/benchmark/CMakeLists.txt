cmake_minimum_required(VERSION 3.14)
project(imagedecoder_benchmark C CXX ASM)
include(FetchContent)

set(CMAKE_BUILD_TYPE Release)

# 16 KB page-size alignment (required by Google Play for Android 15+ devices).
set(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} -Wl,-z,max-page-size=16384")

# ── The benchmark C++ library ────────────────────────────────────────────────
# The CMake target is called "imagedecoder" because every codec subdirectory
# (lcms/, libjpeg-turbo/, libpng/, libwebp/, libjxl/) hardcodes that name in
# their target_link_libraries / target_include_directories calls.
# We set OUTPUT_NAME so the produced .so is libimagedecoder_cpp.so, which can
# coexist with the Rust libimagedecoder.so at runtime.

add_library(imagedecoder SHARED
  ../java_stream.cpp
  ../java_wrapper_benchmark.cpp
  ../borders.cpp
  ../row_convert.cpp
)
set_target_properties(imagedecoder PROPERTIES OUTPUT_NAME "imagedecoder_cpp")

target_include_directories(imagedecoder PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/..)

# Share the _deps download cache with the heifcodec CMake project.
get_filename_component(deps "../../_deps" REALPATH BASE_DIR "${CMAKE_BINARY_DIR}")
set(FETCHCONTENT_BASE_DIR ${deps})

# ── lcms ──────────────────────────────────────────────────────────────────────
add_subdirectory(../lcms ${CMAKE_CURRENT_BINARY_DIR}/lcms-build)

# ── JPEG ──────────────────────────────────────────────────────────────────────
add_subdirectory(../libjpeg-turbo ${CMAKE_CURRENT_BINARY_DIR}/libjpeg-turbo-build)
add_definitions(-DHAVE_LIBJPEG)
target_sources(imagedecoder PRIVATE ../decoder_jpeg.cpp)

# ── PNG ───────────────────────────────────────────────────────────────────────
add_subdirectory(../libpng ${CMAKE_CURRENT_BINARY_DIR}/libpng-build)
add_definitions(-DHAVE_LIBPNG)
target_sources(imagedecoder PRIVATE ../decoder_png.cpp)

# ── WebP ──────────────────────────────────────────────────────────────────────
add_subdirectory(../libwebp ${CMAKE_CURRENT_BINARY_DIR}/libwebp-build)
add_definitions(-DHAVE_LIBWEBP)
target_sources(imagedecoder PRIVATE ../decoder_webp.cpp)

# ── HEIF/AVIF ────────────────────────────────────────────────────────────────
# Both the C++ and Rust decoders use libheif under the hood, so the comparison
# is mainly JNI/wrapper overhead.  We link against the pre-built
# libheifcodec.so (libheif v1.21.2 + libde265 + dav1d) that is already built
# by the heifcodec CMake project and copied to native-libs/{ABI}/.
set(HEIFCODEC_DIR "${CMAKE_CURRENT_SOURCE_DIR}/../../rust/native-libs/${ANDROID_ABI}")
add_definitions(-DHAVE_LIBHEIF)
target_sources(imagedecoder PRIVATE ../decoder_heif.cpp)
target_link_libraries(imagedecoder "${HEIFCODEC_DIR}/libheifcodec.so")

# We still need libheif headers.  FetchContent will re-use the sources already
# downloaded by the heifcodec project (shared FETCHCONTENT_BASE_DIR).
FetchContent_Declare(libheif
  GIT_REPOSITORY  https://github.com/strukturag/libheif.git
  GIT_TAG         v1.21.2
  BINARY_DIR      libheif-build
  SUBBUILD_DIR    libheif-subbuild
)
FetchContent_GetProperties(libheif)
if(NOT libheif_POPULATED)
  FetchContent_Populate(libheif)
  # We only need the headers — do NOT add_subdirectory (the heifcodec project
  # already built the library).
endif()

# heif_version.h is normally generated by libheif's cmake configure step.
# Since we only populated the source (no configure), generate it ourselves.
set(HEIF_VERSION_GEN_DIR "${CMAKE_CURRENT_BINARY_DIR}/generated/libheif")
file(MAKE_DIRECTORY "${HEIF_VERSION_GEN_DIR}")
file(WRITE "${HEIF_VERSION_GEN_DIR}/heif_version.h"
"#ifndef LIBHEIF_HEIF_VERSION_H\n"
"#define LIBHEIF_HEIF_VERSION_H\n"
"#define LIBHEIF_NUMERIC_VERSION ((1<<24) | (21<<16) | (2<<8) | 0)\n"
"#define LIBHEIF_VERSION \"1.21.2\"\n"
"#define LIBHEIF_PLUGIN_DIRECTORY \"\"\n"
"#endif\n"
)

# <libheif/heif.h> is at api/libheif/heif.h → include path = api/
# <libheif/heif_version.h> is generated → include path = generated/
target_include_directories(imagedecoder PRIVATE
  "${libheif_SOURCE_DIR}/libheif/api"
  "${CMAKE_CURRENT_BINARY_DIR}/generated"
)

# ── JXL ───────────────────────────────────────────────────────────────────────
add_subdirectory(../libjxl ${CMAKE_CURRENT_BINARY_DIR}/libjxl-build)
add_definitions(-DHAVE_LIBJXL)
target_sources(imagedecoder PRIVATE ../decoder_jxl.cpp)

target_link_libraries(imagedecoder android jnigraphics log)
